/**
 * A simple panel that adds the ability to catch mouse and keyboard events.
 * Use instead of FocusPanel.
 */

@SuppressWarnings("deprecation")

    SourcesClickEvents, SourcesMouseEvents, SourcesMouseWheelEvents,
